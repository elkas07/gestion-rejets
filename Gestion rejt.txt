                console.error(' Erreur chargement statut KYC:', error);

        //  CORRECTION 1/2  - Charger les commissions avec calcul automatique
        async function loadCommissions() {
            console.log('--- Tentative de chargement des commissions ---');
            
            if (!isSupabaseConnected || !window.supabaseClient) {
                console.log(' Chargement commissions: mode hors ligne');
                commissions = [];
                displayCommissions();
                return;
            }

            try {
                // Charger les projets pour calculer les commissions
                await loadProjects();
                
                // Si aucun projet n'est charg, crer des commissions bases sur les projets
                if (projects.length > 0) {
                    commissions = projects.map(project => {
                        const commissionRate = commissionSettings.defaultRate;
                        const collectedAmount = project.collected || 0;
                        const commissionAmount = collectedAmount * (commissionRate / 100);
                        
                        return {
                            id: project.id,
                            project_id: project.id,
                            project_title: project.name,
                            rate: commissionRate,
                            amount: commissionAmount,
                            collected_amount: collectedAmount,
                            status: project.status === 'completed' ? 'paid' : 'pending',
                            created_at: project.created_at || new Date().toISOString()
                        };
                    });
                    
                    console.log(' Commissions calcules:', commissions.length);
                } else {
                    commissions = [];
                }
                
            } catch (error) {
                console.error(' Erreur critique chargement commissions:', error);
                commissions = [];
            }

            // Appelle la fonction d'affichage aprs chaque chargement
            displayCommissions();
        }

        //  CORRECTION AMLIORE  - Charger les utilisateurs avec pagination
        async function loadUsers(page = 1, pageSize = 50, search = '', role = '', status = '', kycStatus = '') {
            if (!isSupabaseConnected || !window.supabaseClient) {
                console.log(' Chargement utilisateurs: mode hors ligne');
                users = getDefaultUsers();
                totalUserCount = users.length;
                updateAdminUsersList();
                updatePaginationControls();
                return;
            }

            try {
                const loadingIndicator = document.getElementById('usersLoading');
                if (loadingIndicator) {
                    loadingIndicator.classList.add('active');
                }
                
                // Construire la requte avec pagination
                let query = window.supabaseClient
                    .from('members')
                    .select('*', { count: 'exact' });
                
                // Appliquer les filtres
                if (search) {
                    query = query.or(`name.ilike.%${search}%,email.ilike.%${search}%,phone.ilike.%${search}%`);
                }
                
                if (role) {
                    query = query.eq('role', role);
                }
                
                if (status) {
                    query = query.eq('status', status);
                }
                
                // Appliquer le tri
                const sortField = advancedFilters.sortBy.replace('_desc', '').replace('_asc', '');
                const sortDirection = advancedFilters.sortBy.includes('desc') ? false : true;
                query = query.order(sortField, { ascending: sortDirection });
                
                // Calculer les limites de pagination
                const from = (page - 1) * pageSize;
                const to = from + pageSize - 1;
                
                const { data, error, count } = await query.range(from, to);
                
                if (error) throw error;
                
                users = data || [];
                totalUserCount = count || 0;
                
                // Mettre  jour l'affichage
                updateAdminUsersList();
                updatePaginationControls();
                
                if (loadingIndicator) {
                    loadingIndicator.classList.remove('active');
                }
                
                console.log(` Utilisateurs chargs: page ${page}, taille ${pageSize}, total ${totalUserCount}`);
                
            } catch (error) {
                console.error(' Erreur chargement utilisateurs:', error);
                users = getDefaultUsers();
                totalUserCount = users.length;
                updateAdminUsersList();
                updatePaginationControls();
                
                const loadingIndicator = document.getElementById('usersLoading');
                if (loadingIndicator) {
                    loadingIndicator.classList.remove('active');
                }
            }
        }

            // Numros de page
            const maxVisiblePages = 5;
            let startPage = Math.max(1, currentUserPage - Math.floor(maxVisiblePages / 2));
            let endPage = Math.min(totalPages, startPage + maxVisiblePages - 1);
            
            if (endPage - startPage + 1 < maxVisiblePages) {
                startPage = Math.max(1, endPage - maxVisiblePages + 1);
            }
            
            for (let i = startPage; i <= endPage; i++) {
                if (i === currentUserPage) {
                    paginationHTML += `
                        <button class="pagination-number active" onclick="changeUserPage(${i})">
                            ${i}
                        </button>
                    `;
                } else {
                    paginationHTML += `
                        <button class="pagination-number" onclick="changeUserPage(${i})">
                            ${i}
                        </button>
                    `;
                }
            }
            
            // Bouton suivant
            if (currentUserPage < totalPages) {
                paginationHTML += `
                    <button class="pagination-number" onclick="changeUserPage(${currentUserPage + 1})">
                        <i class="fas fa-chevron-right"></i>
                    </button>
                `;
            }
            
            paginationNumbers.innerHTML = paginationHTML;
        }

        // Authentication functions
        async function checkAuthState() {
            if (!isSupabaseConnected || !window.supabaseClient) {
                console.log(' Vrification auth: mode hors ligne');
                return;
            }

            try {
                const { data: { session }, error } = await window.supabaseClient.auth.getSession();
                if (error) throw error;
                
                if (session) {
                    currentUser = session.user;
                    await handleUserLogin(session.user);
                }
            } catch (error) {
                console.error('Erreur vrification auth:', error);
            }
        }

        async function login(email, password) {
            if (!isSupabaseConnected || !window.supabaseClient) {
                console.log(' Connexion: mode hors ligne');
                showNotification('Mode hors ligne - Connexion simule', 'warning');
                
                currentUser = {
                    email: email,
                    user_metadata: { name: email.split('@')[0] }
                };
                
                updateUI();
                showNotification('Connexion simule (mode hors ligne)');
                closeModal('loginModal');
                
                if (email === 'elkas07@live.fr') {
                    showSection('adminDashboard');
                    updateAdminDashboard();
                } else {
                    showSection('memberDashboard');
                    updateMemberDashboard();
                }
                return;
            }

            try {
                const { data, error } = await window.supabaseClient.auth.signInWithPassword({
                    email: email,
                    password: password,
                });
                
                if (error) throw error;
                
                currentUser = data.user;
                await handleUserLogin(data.user);
                
            } catch (error) {
                console.error(' Erreur connexion:', error);
                showNotification('Email ou mot de passe incorrect', 'error');
            }
        }

        async function register(name, email, password, phone) {
            if (!isSupabaseConnected || !window.supabaseClient) {
                console.log(' Inscription: mode hors ligne');
                showNotification('Mode hors ligne - Inscription simule', 'warning');
                
                currentUser = {
                    email: email,
                    user_metadata: { name: name, phone: phone }
                };
                
                updateUI();
                showNotification('Inscription simule (mode hors ligne)');
                closeModal('registrationModal');
                showSection('memberDashboard');
                updateMemberDashboard();
                return;
            }

            try {
                const { data, error } = await window.supabaseClient.auth.signUp({
                    email: email,
                    password: password,
                    options: {
                        data: {
                            name: name,
                            phone: phone
                        }
                    }
                });
                
                if (error) throw error;
                
                if (data.user) {
                    // Crer l'utilisateur dans la table members
                    const { error: insertError } = await window.supabaseClient
                        .from('members')
                        .insert({
                            email: email,
                            name: name,
                            phone: phone,
                            role: 'investor',
                            status: 'active',
                            balance: 0,
                            total_invested: 0
                        });
                    
                    if (insertError) {
                        console.warn('Avertissement insertion membre:', insertError);
                    }
                }
                
                showNotification('Inscription russie ! Vrifiez votre email.');
                closeModal('registrationModal');
                
            } catch (error) {
                console.error(' Erreur inscription:', error);
                showNotification('Erreur lors de l\'inscription: ' + error.message, 'error');
            }
        }

        //  NOUVELLE FONCTION  - Initialisation de l'upload de document KYC
        function initKYCDocumentUpload() {
            const uploadArea = document.getElementById('documentUploadArea');
            const fileInput = document.getElementById('kycDocumentUpload');
            const documentTypeRadios = document.querySelectorAll('input[name="documentType"]');
            
            // Grer le changement de type de document
            documentTypeRadios.forEach(radio => {
                radio.addEventListener('change', function() {
                    selectedDocumentType = this.value;
                    updateDocumentTitle();
                    // Rinitialiser le fichier slectionn
                    selectedDocumentFile = null;
                    document.getElementById('documentPreviewContainer').style.display = 'none';
                    document.getElementById('submitKYCBtn').disabled = true;
                    document.getElementById('uploadProgressContainer').style.display = 'none';
                });
            });
            
            // Grer le clic sur la zone d'upload
            uploadArea.addEventListener('click', function() {
                fileInput.click();
            });
            
            // Grer le drag and drop
            uploadArea.addEventListener('dragover', function(e) {
                e.preventDefault();
                this.style.borderColor = 'var(--accent-color)';
                this.style.backgroundColor = '#fff5f0';
            });
            
            uploadArea.addEventListener('dragleave', function(e) {
                e.preventDefault();
                this.style.borderColor = '#cbd5e0';
                this.style.backgroundColor = '#f8fafc';
            });
            
            uploadArea.addEventListener('drop', function(e) {
                e.preventDefault();
                this.style.borderColor = '#cbd5e0';
                this.style.backgroundColor = '#f8fafc';
                
                if (e.dataTransfer.files.length > 0) {
                    handleDocumentSelection(e.dataTransfer.files[0]);
                }
            });
            
            // Grer la slection de fichier
            fileInput.addEventListener('change', function(e) {
                if (e.target.files.length > 0) {
                    handleDocumentSelection(e.target.files[0]);
                }
            });
            
            // Mettre  jour le titre initial
            updateDocumentTitle();
        }
        
        function updateDocumentTitle() {
            const titles = {
                'passport': 'Soumettre votre passeport',
                'nni': 'Soumettre votre carte d\'identit nationale (NNI)',
                'driver_license': 'Soumettre votre permis de conduire'
            };
            document.getElementById('selectedDocumentTitle').textContent = titles[selectedDocumentType] || 'Soumettre votre document';
        }
        
        function handleDocumentSelection(file) {
            // Vrifier la taille du fichier (max 5MB)
            if (file.size > 5 * 1024 * 1024) {
                showNotification('Le fichier est trop volumineux (max. 5MB)', 'error');
                return;
            }
            
            // Vrifier le type de fichier
            const validTypes = ['image/jpeg', 'image/jpg', 'image/png', 'application/pdf'];
            if (!validTypes.includes(file.type)) {
                showNotification('Format de fichier non support. Utilisez JPG, PNG ou PDF.', 'error');
                return;
            }
            
            selectedDocumentFile = file;
            
            // Afficher l'aperu
            const previewContainer = document.getElementById('documentPreviewContainer');
            const documentName = document.getElementById('documentName');
            const previewImg = document.getElementById('documentPreviewImg');
            
            documentName.textContent = file.name;
            
            if (file.type.startsWith('image/')) {
                const reader = new FileReader();
                reader.onload = function(e) {
                    previewImg.src = e.target.result;
                    previewImg.style.display = 'block';
                };
                reader.readAsDataURL(file);
            } else {
                previewImg.style.display = 'none';
            }
            
            previewContainer.style.display = 'block';
            document.getElementById('submitKYCBtn').disabled = false;
        }

        //  CORRECTION CRITIQUE  - Soumettre un document KYC
        async function submitKYCDocument() {
            if (!currentUser) {
                showNotification('Veuillez vous connecter', 'error');
                return;
            }

            if (!selectedDocumentFile) {
                showNotification('Veuillez slectionner un document', 'error');
                return;
            }

            if (!isSupabaseConnected || !window.supabaseClient) {
                showNotification('Mode hors ligne - Impossible de soumettre le document', 'warning');
                return;
            }

            try {
                const submitBtn = document.getElementById('submitKYCBtn');
                submitBtn.disabled = true;
                submitBtn.innerHTML = '<i class="fas fa-spinner fa-spin"></i> Traitement en cours...';

                const progressContainer = document.getElementById('uploadProgressContainer');
                const progressBar = document.getElementById('uploadProgressBar');
                const progressText = document.getElementById('uploadProgressText');
                const progressPercentage = document.getElementById('uploadPercentage');

                progressContainer.style.display = 'block';
                progressText.textContent = 'Tlchargement du document...';
                progressBar.style.width = '10%';
                progressPercentage.textContent = '10%';

                // 1. Rcuprer ou crer le membre
                const { data: memberData, error: memberError } = await window.supabaseClient
                    .from('members')
                    .select('id')
                    .eq('email', currentUser.email)
                    .maybeSingle();

                if (memberError) {
                    throw new Error('Erreur rcupration membre: ' + memberError.message);
                }

                let memberId;
                if (!memberData) {
                    // Crer le membre
                    const { data: newMember, error: createError } = await window.supabaseClient
                        .from('members')
                        .insert({
                            email: currentUser.email,
                            name: currentUser.user_metadata?.name || currentUser.email.split('@')[0],
                            phone: currentUser.user_metadata?.phone || '',
                            role: 'investor',
                            status: 'active',
                            balance: 0,
                            total_invested: 0
                        })
                        .select('id')
                        .single();

                    if (createError) throw createError;
                    memberId = newMember.id;
                } else {
                    memberId = memberData.id;
                }

                progressBar.style.width = '30%';
                progressPercentage.textContent = '30%';
                progressText.textContent = 'Prparation de l\'upload...';

                // 2. Uploader vers le bucket kyc_documents
                const fileName = `kyc_${memberId}_${Date.now()}.${selectedDocumentFile.name.split('.').pop()}`;
                
                const { error: uploadError } = await window.supabaseClient.storage
                    .from('kyc_documents')  //  Note: le nom du bucket est sensible  la casse
                    .upload(fileName, selectedDocumentFile, {
                        cacheControl: '3600',
                        upsert: false
                    });

                if (uploadError) {
                    console.error(' Erreur upload storage:', uploadError);
                    throw new Error('Erreur lors du tlchargement du document: ' + uploadError.message);
                }

                progressBar.style.width = '70%';
                progressPercentage.textContent = '70%';
                progressText.textContent = 'Rcupration de l\'URL...';

                // 3. Rcuprer l'URL publique
                const { data: urlData } = window.supabaseClient.storage
                    .from('kyc_documents')
                    .getPublicUrl(fileName);

                const fileUrl = urlData.publicUrl;

                progressBar.style.width = '90%';
                progressPercentage.textContent = '90%';
                progressText.textContent = 'Enregistrement dans la base de donnes...';

                // 4. Crer l'enregistrement KYC
                const kycData = {
                    user_id: memberId,
                    document_type: selectedDocumentType,
                    status: 'pending', //  CORRECTION : Statut dfini  'pending' immdiatement
                    submitted_at: new Date().toISOString(),
                    updated_at: new Date().toISOString(),
                    documents: [
                        {
                            type: selectedDocumentType,
                            file_name: selectedDocumentFile.name,
                            file_url: fileUrl,
                            file_size: selectedDocumentFile.size,
                            uploaded_at: new Date().toISOString()
                        }
                    ]
                };

                const { error: kycError } = await window.supabaseClient
                    .from('kyc_verifications')
                    .insert([kycData]);

                if (kycError) {
                    console.error(' Erreur enregistrement KYC:', kycError);
                    
                    // Nettoyer le fichier upload en cas d'erreur
                    try {
                        await window.supabaseClient.storage
                            .from('kyc_documents')
                            .remove([fileName]);
                    } catch (cleanupError) {
                        console.warn(' Erreur nettoyage fichier:', cleanupError);
                    }
                    
                    throw new Error('Erreur lors de la soumission du KYC: ' + kycError.message);
                }

                progressBar.style.width = '100%';
                progressPercentage.textContent = '100%';
                progressText.textContent = 'Document soumis avec succs !';

                //  CORRECTION : Mettre  jour le statut local IMMDIATEMENT
                currentKYCStatus = 'pending';
                await loadUserKYCStatus(); // Recharger pour tre sr
                updateKYCStatusDisplay();

                // Recharger les vrifications KYC
                await loadKYCVerifications();

                setTimeout(() => {
                    showNotification('Document KYC soumis avec succs ! Vrification en cours.');
                    closeModal('kycPortalModal');
                    
                    if (submitBtn) {
                        submitBtn.disabled = false;
                        submitBtn.innerHTML = '<i class="fas fa-paper-plane"></i> Soumettre pour vrification';
                    }

                    if (progressContainer) {
                        progressContainer.style.display = 'none';
                    }
                }, 1000);

            } catch (error) {
                console.error(' Erreur soumission KYC:', error);

                const submitBtn = document.getElementById('submitKYCBtn');
                if (submitBtn) {
                    submitBtn.disabled = false;
                    submitBtn.innerHTML = '<i class="fas fa-paper-plane"></i> Soumettre pour vrification';
                }

                const progressContainer = document.getElementById('uploadProgressContainer');
                if (progressContainer) {
                    progressContainer.style.display = 'none';
                }

                showNotification('Erreur lors de la soumission: ' + error.message, 'error');
            }
        }

        //  CORRECTION CRITIQUE  - Approuver une vrification KYC
        async function approveKYC(kycId = null) {
            console.log(" Lancement de la validation RELLE..."); 
            
            // 1. Identification du KYC concern
            const kyc = kycId ? kycVerifications.find(k => k.id === kycId) : currentViewingKYC;
            
            if (!kyc) {
                console.error(" Impossible de trouver les donnes du KYC");
                showNotification('KYC non trouv', 'error');
                return;
            }

            if (!confirm('CONFIRMATION : Voulez-vous enregistrer cette validation en BASE DE DONNES ?')) {
                return;
            }

            try {
                // 2. Prparation des donnes
                const noteElement = document.getElementById('kycAdminNote');
                const note = noteElement ? noteElement.value : null;
                const now = new Date().toISOString();
                
                // 3. Envoi  Supabase
                const { error } = await window.supabaseClient
                    .from('kyc_verifications')
                    .update({
                        status: 'verified',
                        updated_at: now,
                        reviewed_at: now,
                        reviewer_notes: note || null
                    })
                    .eq('id', kyc.id);

                if (error) throw error;

                // 4. MISE  JOUR LOCALE FORCE
                const index = kycVerifications.findIndex(k => k.id === kyc.id);
                if (index !== -1) {
                    kycVerifications[index].status = 'verified';
                }

                // 5. Notification de succs rel
                showNotification(' BASE DE DONNES MISE  JOUR : KYC valid');
                
                // 6. Rafrachissement complet des donnes et de l'interface
                await loadKYCVerifications();
                
                updateAdminKYCList();

                // 7. Nettoyage et fermeture du panneau de dtails
                const detailsContainer = document.getElementById('kycDetailsContainer');
                if (detailsContainer) {
                    detailsContainer.style.display = 'none';
                }
                if (noteElement) {
                    noteElement.value = '';
                }

                console.log(' Validation termine pour:', kyc.id);

            } catch (error) {
                console.error(' ERREUR SUPABASE:', error);
                showNotification('Erreur relle : ' + error.message, 'error');
            }
        }
        
        //  NOUVELLE FONCTION  - Rejeter une vrification KYC
        async function rejectKYC(kycId = null) {
            const kyc = kycId ? kycVerifications.find(k => k.id === kycId) : currentViewingKYC;
            if (!kyc) return;

            if (!confirm('tes-vous sr de vouloir rejeter cette vrification KYC ?')) {
                return;
            }

            try {
                const note = document.getElementById('kycAdminNote').value;
                const reason = prompt('Veuillez indiquer la raison du rejet (optionnel):');
                
                const { error } = await window.supabaseClient
                    .from('kyc_verifications')
                    .update({
                        status: 'rejected',
                        updated_at: new Date().toISOString(),
                        reviewed_at: new Date().toISOString(),
                        rejection_reason: reason || null,
                        reviewer_notes: note || null
                    })
                    .eq('id', kyc.id);

                if (error) throw error;

                showNotification('Vrification KYC rejete avec succs');
                await loadKYCVerifications();
                updateAdminKYCList();
                document.getElementById('kycDetailsContainer').style.display = 'none';
                document.getElementById('kycAdminNote').value = '';

            } catch (error) {
                console.error(' Erreur rejet KYC:', error);
                showNotification('Erreur lors du rejet: ' + error.message, 'error');
            }
        }

        //  CORRECTION MAJEURE  - Fonction validateTransaction
        async function validateTransaction(transactionId) {
            if (!isSupabaseConnected || !window.supabaseClient) {
                showNotification('Mode hors ligne - Impossible de valider les transactions', 'warning');
                return;
            }

            try {
                console.log(' Validation de la transaction:', transactionId);
                
                // 1. Rcuprer la transaction
                const { data: transaction, error: transactionError } = await window.supabaseClient
                    .from('transactions')
                    .select('*')
                    .eq('id', transactionId)
                    .single();

                if (transactionError) {
                    console.error(' Transaction non trouve:', transactionError);
                    showNotification('Transaction non trouve.', 'error');
                    return;
                }

                // 2. Rcuprer le membre
                const { data: member, error: memberError } = await window.supabaseClient
                    .from('members')
                    .select('*')
                    .eq('id', transaction.member_id)
                    .single();

                if (memberError) {
                    console.error(' Erreur rcupration membre:', memberError);
                    showNotification('Erreur: Impossible de rcuprer le membre.', 'error');
                    return;
                }

                // 3. Traiter selon le type de transaction
                if (transaction.type === 'deposit') {
                    // Mise  jour du solde du membre
                    const newBalance = (member.balance || 0) + transaction.amount;
                    const { error: updateError } = await window.supabaseClient
                        .from('members')
                        .update({ balance: newBalance })
                        .eq('id', member.id);

                    if (updateError) {
                        console.error(' Erreur mise  jour solde:', updateError);
                        throw updateError;
                    }

                    showNotification(`Dpt de ${formatCFA(transaction.amount)} valid ! Solde mis  jour.`);

                } else if (transaction.type === 'investment') {
                    // Vrifier le solde
                    if (member.balance < transaction.amount) {
                        showNotification('Solde insuffisant pour valider cet investissement', 'error');
                        return;
                    }

                    // Mettre  jour le solde et total investi du membre
                    const newBalance = member.balance - transaction.amount;
                    const { error: updateBalanceError } = await window.supabaseClient
                        .from('members')
                        .update({ 
                            balance: newBalance,
                            total_invested: (member.total_invested || 0) + transaction.amount
                        })
                        .eq('id', member.id);

                    if (updateBalanceError) throw updateBalanceError;

                    // Mettre  jour le projet
                    const project = projects.find(p => p.id == transaction.project_id);
                    if (project) {
                        const newCollectedAmount = project.collected + transaction.amount;
                        const { error: updateProjectError } = await window.supabaseClient
                            .from('projects')
                            .update({ collected: newCollectedAmount })
                            .eq('id', transaction.project_id);

                        if (updateProjectError) throw updateProjectError;

                        project.collected = newCollectedAmount;
                        
                        // Vrifier si le projet est complet
                        if (newCollectedAmount >= project.target) {
                            project.status = 'completed';
                        }
                    }

                    // Crer la contribution
                    const { error: contributionError } = await window.supabaseClient
                        .from('contributions')
                        .insert([
                            {
                                member_id: member.id,
                                project_id: transaction.project_id,
                                amount: transaction.amount,
                                payment_method: transaction.payment_method,
                                status: 'completed',
                                created_at: new Date().toISOString()
                            }
                        ]);

                    if (contributionError) throw contributionError;

                    showNotification(`Investissement de ${formatCFA(transaction.amount)} valid ! Projet mis  jour.`);
                }

                // 4. Mettre  jour le statut de la transaction
               const { error: updateStatusError } = await window.supabaseClient
                    .from('transactions')
                    .update({ 
                        status: 'validated',
                        validated_at: new Date().toISOString()
                    })
                    .eq('id', transactionId);

                if (updateStatusError) throw updateStatusError;

                // 5. Retirer la ligne du tableau
                removeTransactionRow(transactionId);

                // 6. Attendre un court instant puis mettre  jour les donnes
                setTimeout(async () => {
                    await loadPendingTransactions();
                    await loadProjects();
                    await loadContributions();
                    await loadUsers();
                    
                    // 7. Mettre  jour les dashboards
                    if (currentUser && currentUser.email === 'elkas07@live.fr') {
                        updateAdminDashboard();
                    } else if (currentUser) {
                        updateMemberDashboard();
                    }
                }, 1000); 
                
                console.log(' Transaction valide avec succs:', transactionId);
                
            } catch (error) {
                console.error(' Erreur validation transaction:', error);
                showNotification('Erreur lors de la validation: ' + error.message, 'error');
            }
        }

        //  NOUVELLE FONCTION  - Rejeter une transaction
        async function rejectTransaction(transactionId) {
            if (!isSupabaseConnected || !window.supabaseClient) {
                showNotification('Mode hors ligne - Impossible de rejeter les transactions', 'warning');
                return;
            }

            if (confirm('tes-vous sr de vouloir refuser cette transaction ?')) {
                try {
                    console.log(' Rejet de la transaction:', transactionId);
                    
                    const { error } = await window.supabaseClient
                        .from('transactions')
                        .update({ 
                            status: 'rejected',
                            rejected_at: new Date().toISOString()
                        })
                        .eq('id', transactionId);

                    if (error) throw error;

                    console.log(` Transaction ${transactionId} rejete avec succs.`);
                    
                    // Supprimer la ligne du tableau
                    removeTransactionRow(transactionId);
                    
                    // Mettre  jour les donnes
                    pendingTransactions = pendingTransactions.filter(t => t.id !== transactionId);
                    
                    // Mettre  jour le dashboard
                    updateAdminDashboard();
                    
                    showNotification(`Transaction ${transactionId} rejete avec succs.`);
                    
                } catch (error) {
                    console.error(' Erreur rejet transaction:', error);
                    showNotification('Erreur lors du rejet: ' + error.message, 'error');
                }
            }
        }

        //  NOUVELLE FONCTION  - Retirer une ligne du tableau
        function removeTransactionRow(transactionId) {
            const tableBody = document.getElementById('adminPendingTransactions');
            if (!tableBody) return;
            
            const rows = tableBody.querySelectorAll('tr');
            rows.forEach(row => {
                const validateBtn = row.querySelector('.btn-validate');
                if (validateBtn && validateBtn.onclick && 
                    validateBtn.onclick.toString().includes(`'${transactionId}'`)) {
                    row.remove();
                }
            });
            
            // Vrifier s'il reste des transactions
            const remainingRows = tableBody.querySelectorAll('tr');
            if (remainingRows.length === 0) {
                tableBody.innerHTML = '<tr><td colspan="7" style="text-align: center;">Aucune transaction en attente</td></tr>';
            }
        }

        // Menu mobile
        function setupMobileMenu() {
            const mobileMenuBtn = document.getElementById('mobileMenuBtn');
            const mobileMenuOverlay = document.getElementById('mobileMenuOverlay');
            const navLinks = document.getElementById('navLinks');

            if (mobileMenuBtn && mobileMenuOverlay && navLinks) {
                mobileMenuBtn.addEventListener('click', function() {
                    this.classList.toggle('active');
                    navLinks.classList.toggle('active');
                    mobileMenuOverlay.classList.toggle('active');
                    document.body.style.overflow = this.classList.contains('active') ? 'hidden' : '';
                });

                mobileMenuOverlay.addEventListener('click', function() {
                    mobileMenuBtn.classList.remove('active');
                    navLinks.classList.remove('active');
                    this.classList.remove('active');
                    document.body.style.overflow = '';
                });
            }
        }

        function closeMobileMenu() {
            const mobileMenuBtn = document.getElementById('mobileMenuBtn');
            const mobileMenuOverlay = document.getElementById('mobileMenuOverlay');
            const navLinks = document.getElementById('navLinks');

            if (mobileMenuBtn && mobileMenuOverlay && navLinks) {
                mobileMenuBtn.classList.remove('active');
                navLinks.classList.remove('active');
                mobileMenuOverlay.classList.remove('active');
                document.body.style.overflow = '';
            }
        }

        // Notification
        function showNotification(message, type = 'success') {
            const notification = document.getElementById('notification');
            const messageElement = document.getElementById('notification-message');
            const icon = notification.querySelector('i');
            
            messageElement.textContent = message;
            notification.className = 'notification ' + type;
            
            if (type === 'error') {
                icon.className = 'fas fa-exclamation-circle';
            } else if (type === 'warning') {
                icon.className = 'fas fa-exclamation-triangle';
            } else {
                icon.className = 'fas fa-check-circle';
            }
            
            notification.style.display = 'flex';
            
            setTimeout(() => {
                notification.style.display = 'none';
            }, 5000);
        }
